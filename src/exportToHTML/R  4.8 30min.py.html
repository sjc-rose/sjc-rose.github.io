4    from sklearn.ensemble import RandomForestRegressor

12   def forecast_30_minutes():
13       file_path = '第四周大数据分析作业.xlsx'
14       try:
15           # 读取数据并排序
16           df = pd.read_excel(file_path).sort_values('datetime')
17   
18           # 设定预测起点：2019-04-08 05:04:00
19           start_time = pd.to_datetime('2019-04-08 05:04:00')
20   
21           # 准备训练集：使用起点之前的所有数据
22           train_df = df[df['datetime'] <= start_time].copy()
23   
24           print(f"✅ 训练数据准备就绪，截止时间: {start_time}")
25   
26           # 2. 特征工程
27           # 使用过去 15 分钟的 [close, open, high] 预测下一分钟的 close
28           window_size = 15
29           feature_cols = ['close', 'open', 'high']
30           train_data = train_df[feature_cols].values
31   
32           X, y = [], []
33           for i in range(len(train_data) - window_size):
34               X.append(train_data[i: i + window_size].flatten())
35               y.append(train_data[i + window_size, 0])  # 预测目标是 close
36   
37           X, y = np.array(X), np.array(y)
38   
39           # 3. 训练模型
40           model = RandomForestRegressor(n_estimators=100, random_state=42)
41           model.fit(X, y)
42   
43           # 4. 滚动预测未来 30 分钟
44           prediction_steps = 30
45           current_window = train_data[-window_size:].tolist()
46   
47           forecast_prices = []
48           forecast_times = []
49           curr_time = start_time
50   
51           for _ in range(prediction_steps):
52               # 获取当前输入窗口
53               input_x = np.array(current_window[-window_size:]).flatten().reshape(1, -1)
54               # 预测下一分钟价格
55               pred_close = model.predict(input_x)[0]
56   
57               # 记录结果
58               forecast_prices.append(pred_close)
59               curr_time += timedelta(minutes=1)
60               forecast_times.append(curr_time)
61   
62               # 将预测值作为下一分钟的输入（假设 open 和 high 与 close 接近）
63               current_window.append([pred_close, pred_close, pred_close])
64   
65           # 5. 可视化对比图
66           plt.figure(figsize=(12, 6))
67   
68           # 画出预测点之前的真实走势（展示最后 60 分钟）
69           recent_history = train_df.tail(60)
70           plt.plot(recent_history['datetime'], recent_history['close'],
71                    label='历史真实价格', color='#1f77b4', linewidth=2)
72   
73           # 画出预测的 30 分钟走势
74           plt.plot(forecast_times, forecast_prices,
75                    label='未来 30 分钟预测', color='#e67e22', linestyle='--', marker='o', markersize=4)
76   
77           # 装饰图表
78           plt.axvline(x=start_time, color='red', linestyle=':', label='预测起点 (05:04)')
79           plt.title('以太币 (ETH) 未来 30 分钟价格预测 (基于 4/8 05:04 数据)', fontsize=14)
80           plt.xlabel('时间')
81           plt.ylabel('价格 (USD)')
82           plt.legend()
83           plt.grid(True, alpha=0.3)
84   
85           # 设置横坐标格式
86           plt.gcf().autofmt_xdate()
87   
88           plt.show()
89   
90           print("-" * 30)
91           print(f"预测完成！起点价格: {train_data[-1, 0]:.2f}")
92           print(f"30分钟后预计价格: {forecast_prices[-1]:.2f}")
93           print(f"预计总变动: {forecast_prices[-1] - train_data[-1, 0]:+.2f}")
94           print("-" * 30)
95   
96       except Exception as e:
97           print(f"❌ 运行失败: {e}")
98   
99   
100  if __name__ == "__main__":
101      forecast_30_minutes()
